#include <iostream>
#include <vector>
#include <string>
#include <limits>
#include "sorting.h"
#include "steps.h"

using namespace std;

/* ── Display a vector array ─────────────────────────────── */
void printArray(const vector<int>& arr, const string& label) {
    cout << label << " [ ";
    for (size_t i = 0; i < arr.size(); i++) {
        cout << arr[i];
        if (i < arr.size() - 1) cout << ", ";
    }
    cout << " ]" << endl;
}

/* ── Display sort steps ─────────────────────────────────── */
void printSteps(const vector<vector<int>>& steps) {
    cout << "\n--- Sort Steps ---\n";
    for (size_t i = 0; i < steps.size(); i++) {
        cout << "Step " << (i + 1) << ": [ ";
        for (size_t j = 0; j < steps[i].size(); j++) {
            cout << steps[i][j];
            if (j < steps[i].size() - 1) cout << ", ";
        }
        cout << " ]" << endl;
    }
}

/* ── Get user input array ───────────────────────────────── */
vector<int> getUserArray() {
    int n;
    cout << "Enter number of elements (2-20): ";
    cin >> n;
    while (n < 2 || n > 20) {
        cout << "Invalid. Enter a value between 2 and 20: ";
        cin.clear();
        cin.ignore(numeric_limits<streamsize>::max(), '\n');
        cin >> n;
    }
    vector<int> arr(n);
    cout << "Enter " << n << " integers: ";
    for (int& val : arr) cin >> val;
    return arr;
}

/* ── Main menu ──────────────────────────────────────────── */
int main() {
    cout << "=================================\n";
    cout << " GROUP UNO — Sorting Visualizer\n";
    cout << "=================================\n\n";

    vector<int> arr = getUserArray();
    printArray(arr, "Original array:");

    cout << "\nSelect Algorithm:\n";
    cout << "  1. Bubble Sort\n";
    cout << "  2. Selection Sort\n";
    cout << "  3. Insertion Sort\n";
    cout << "  4. Merge Sort\n";
    cout << "  5. Quick Sort\n";
    cout << "  6. Heap Sort\n";
    cout << "Choice: ";
    int choice; cin >> choice;

    cout << "\nSort order (1=Ascending, 2=Descending): ";
    int orderChoice; cin >> orderChoice;
    bool ascending = (orderChoice == 1);

    /* Run the selected sort and capture steps */
    Sorter sorter;
    StepRecorder recorder;
    sorter.sort(arr, choice, ascending, recorder);

    printArray(arr, "Sorted array: ");
    printSteps(recorder.getSteps());

    cout << "\nTotal steps: " << recorder.count() << endl;
    return 0;
}

Core Algorithms 
/*
 * sorting.h — Core sorting algorithm implementations
 * Each algorithm records intermediate states via StepRecorder.
 */

#pragma once
#include <vector>
#include <algorithm>
#include "steps.h"

class Sorter {
public:
    /* ── Dispatcher ───────────────────────────────────────── */
    void sort(vector<int>& arr, int choice, bool asc, StepRecorder& rec) {
        switch (choice) {
            case 1: bubbleSort(arr, rec);    break;
            case 2: selectionSort(arr, rec); break;
            case 3: insertionSort(arr, rec); break;
            case 4: mergeSort(arr, 0, arr.size()-1, rec); break;
            case 5: quickSort(arr, 0, arr.size()-1, rec); break;
            case 6: heapSort(arr, rec);     break;
        }
        /* Reverse all steps if descending order requested */
        if (!asc) {
            reverse(arr.begin(), arr.end());
            rec.reverseAll();
        }
    }

private:
    /* ── 1. Bubble Sort ───────────────────────────────────── */
    void bubbleSort(vector<int>& arr, StepRecorder& rec) {
        int n = arr.size();
        for (int i = 0; i < n - 1; i++) {
            bool swapped = false;
            for (int j = 0; j < n - i - 1; j++) {
                if (arr[j] > arr[j + 1]) {
                    swap(arr[j], arr[j + 1]);
                    swapped = true;
                }
            }
            rec.record(arr); // record state after each pass
            if (!swapped) break; // early exit optimization
        }
    }

    /* ── 2. Selection Sort ────────────────────────────────── */
    void selectionSort(vector<int>& arr, StepRecorder& rec) {
        int n = arr.size();
        for (int i = 0; i < n - 1; i++) {
            int minIdx = i;
            for (int j = i + 1; j < n; j++) {
                if (arr[j] < arr[minIdx]) minIdx = j;
            }
            if (minIdx != i) {
                swap(arr[i], arr[minIdx]);
                rec.record(arr); // record after each swap
            }
        }
    }

    /* ── 3. Insertion Sort ────────────────────────────────── */
    void insertionSort(vector<int>& arr, StepRecorder& rec) {
        int n = arr.size();
        for (int i = 1; i < n; i++) {
            int key = arr[i];
            int j   = i - 1;
            while (j >= 0 && arr[j] > key) {
                arr[j + 1] = arr[j];
                j--;
            }
            arr[j + 1] = key;
            rec.record(arr); // record after inserting each element
        }
    }

    /* ── 4. Merge Sort ────────────────────────────────────── */
    void merge(vector<int>& arr, int l, int m, int r, StepRecorder& rec) {
        vector<int> L(arr.begin() + l, arr.begin() + m + 1);
        vector<int> R(arr.begin() + m + 1, arr.begin() + r + 1);
        int i = 0, j = 0, k = l;
        while (i < L.size() && j < R.size())
            arr[k++] = (L[i] <= R[j]) ? L[i++] : R[j++];
        while (i < L.size()) arr[k++] = L[i++];
        while (j < R.size()) arr[k++] = R[j++];
        rec.record(arr);
    }

    void mergeSort(vector<int>& arr, int l, int r, StepRecorder& rec) {
        if (l >= r) return;
        int m = (l + r) / 2;
        mergeSort(arr, l, m, rec);
        mergeSort(arr, m + 1, r, rec);
        merge(arr, l, m, r, rec);
    }

    /* ── 5. Quick Sort ────────────────────────────────────── */
    int partition(vector<int>& arr, int lo, int hi, StepRecorder& rec) {
        int pivot = arr[hi], i = lo - 1;
        for (int j = lo; j < hi; j++) {
            if (arr[j] < pivot) swap(arr[++i], arr[j]);
        }
        swap(arr[i + 1], arr[hi]);
        rec.record(arr);
        return i + 1;
    }

    void quickSort(vector<int>& arr, int lo, int hi, StepRecorder& rec) {
        if (lo < hi) {
            int pi = partition(arr, lo, hi, rec);
            quickSort(arr, lo, pi - 1, rec);
            quickSort(arr, pi + 1, hi, rec);
        }
    }

    /* ── 6. Heap Sort ─────────────────────────────────────── */
    void heapify(vector<int>& arr, int n, int i, StepRecorder& rec) {
        int largest = i, l = 2*i+1, r = 2*i+2;
        if (l < n && arr[l] > arr[largest]) largest = l;
        if (r < n && arr[r] > arr[largest]) largest = r;
        if (largest != i) {
            swap(arr[i], arr[largest]);
            heapify(arr, n, largest, rec);
        }
    }

    void heapSort(vector<int>& arr, StepRecorder& rec) {
        int n = arr.size();
        /* Phase 1: Build max-heap */
        for (int i = n/2-1; i >= 0; i--) heapify(arr, n, i, rec);
        rec.record(arr);
        /* Phase 2: Extract elements from heap */
        for (int i = n-1; i > 0; i--) {
            swap(arr[0], arr[i]);
            heapify(arr, i, 0, rec);
            rec.record(arr);
        }
    }
};
steps.h
Step Recorder 
/*
 * steps.h — StepRecorder class
 * Captures every intermediate state of the array during sorting.
 * Used to power the step-through visualizer in the frontend.
 */

#pragma once
#include <vector>
#include <algorithm>

class StepRecorder {
private:
    /* Internal storage: each element is a snapshot of the array */
    vector<vector<int>> steps;

public:
    /* Record current state of the array as a snapshot */
    void record(const vector<int>& arr) {
        steps.push_back(arr);
    }

    /* Get all recorded steps (read-only) */
    const vector<vector<int>>& getSteps() const {
        return steps;
    }

    /* Return total number of recorded steps */
    size_t count() const {
        return steps.size();
    }

    /* Reverse all steps (used for descending order) */
    void reverseAll() {
        for (auto& step : steps) {
            reverse(step.begin(), step.end());
        }
    }

    /* Clear all recorded steps */
    void clear() {
        steps.clear();
    }
};
